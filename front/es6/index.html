函数优化

  • 函数参数默认值

在es6之前，无法给一个函数参数设置默认值，只能采用变通写法

function add(a,b){
	b =  b || 1;
  return a + b;
}
console.log(add(10));

现在可以直接给参数写上默认值

function add2(a,b=1){
	return a+b;
}
console.log(add2(10));
  • 不定参数

用来表示不确定参数个数。…参数名，由…加上一个具名参数标示符组成。具名参数只能放在参数列表的最后，并且有且只有一个不定参数。

function fun(...values){
	console.log(values.length);
}
fun(1,2,3);

  • 箭头函数

    一个参数时：
//以前声明方法
var print = function(obj){
	console.log(obj);
}
//现在可以这样写：
var print = obj => console.log(obj);


多个参数时：
//以前声明方法：
var sum = function(a,b){
	return a+b;
}
//现在可以这样写：
var sum = (a,b) => a+b;


//代码不止一行时
var sum = (a,b) => {
	c = a + b;
	return c + a;
}

对象优化

  • keys(obj):获取对象的所有key形成的数组
  • values(obj):获取对象的所有value形成的数组
  • entries(obj):获取对象的所有key和value形成的二维数组。格式：[[k1,v1],[k2,v2]….]

  • assign(dest,…src):将多个src对象的值拷贝到dest中。

    const person = {
	name: 'jack',
	age: 21,
	language: ['java','js','css']
}
console.log(Object.keys(person));  //["name","age","language"]
console.log(Object.values(person)); //["jack",21,Array(3)]
console.log(Object.entries(person));//[Array(2),Array(2),Array(2)]



const target = {a:1};
const source1 = {b:2};
const source2 = {c:3};
Object.assign(target,source1,source2);
console.log(target); //{a:1,b:2,c:3}

const name = "zhangsan";
//传统
const person1 = {age:age,name:name};
//es6属性名和属性变量名一样，可以省略

reduce

//reduce():为数组中的每个元素依次执行回调函数，回用到上次执行的返回结果
//语法： arr.reduce(callback,[initialValue]);
//callback包含四个参数：
//1.previousValue（上次调用回调函数返回的值，或者是提供的初始值initialValue）
//2.currentValue（数组中当前被处理的元素）
//3.index（当前元素在数组中的索引）
//4.array（调用reduce的数组）
let  arr = ["1","20","-5","3"];
let result = arr.reduce((a,b) => {
	console.log("上一次处理后"+a);	//  1
	console.log("当前正在处理"+b);  //  20
	return a+b;  
});

//指定初始值
let result2 = arr.reduce((a,b) => {
	console.log("上一次处理后"+a);	//  100
	console.log("当前正在处理"+b);  //  1
	return a+b;  
},100);

Promise

一旦有一连串的ajax请求a,b,c,d…后面的请求依赖前面的请求的结果，就需要层层嵌套。这种缩进和层层嵌套的方式，非常容易造成上下文代码混乱。

let p1 = new Promise((resolve,reject) => {
	 //ajax请求，success方法里面调用resolve(data),error方法里面调用reject(error);
});
let p2 = p1.then((obj) => {
//then方法return一个new Promise方便后面继续进行调用
	return new Promise((resolve,reject) => {
	
	});
}).catch();

可以简单进行封装：

function get(url,data){
	return new Promise((resolve,reject) => {
		$.ajax({
			url:url,
			data:data,
			success:function(data){
				resolve(data);
			},
			error:function(err){
				reject(err);
			}
		});
	});
}

模块化

  • export命令用于规定模块的对外接口
  • import命令用于导入其他模块提供的功能

export不仅可以导出对象，一切js变量都可以导出。比如：基本类型变量、函数、数组、对象。

export {xxx,yyy,zzz}

export是什么名字，import就必须也是什么名字。

可以使用export default{ }将js中所有对象导出，这样的话导入的时候就可以随意起名字了。
